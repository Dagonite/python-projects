def archery_game():
    win = GraphWin("Archery game", 500, 500)
    win.setBackground("cyan")
    win.setCoords(0, 0, 1, 1)
    centre = Point(.5, .5)

    ground_rect = Rectangle(Point(-.01, -.01), Point(1.01, .5))
    ground_rect.setFill("green")
    ground_rect.draw(win)

    left_target_stand = Polygon(Point(.02, -.01), Point(.45, .5),
                                Point(.55, .5), Point(.12, -.01))
    left_target_stand.setFill("brown")
    left_target_stand.setWidth(2)
    left_target_stand.draw(win)

    right_target_stand = Polygon(Point(.98, -.01), Point(.55, .5),
                         Point(.45, .5), Point(.88, -.01))
    right_target_stand.setFill("brown")
    right_target_stand.setWidth(2)
    right_target_stand.draw(win)

    draw_circle(win, centre, .3, "blue")
    draw_circle(win, centre, .2, "red")
    draw_circle(win, centre, .1, "yellow")

    grades = ["You're the best of the best", "You're pretty good", "You're average",
    "You're below average", "You're bad at this"]

    score = 0
    score_text = Text(Point(.5, .03), "Score: {0}".format(score))
    score_text.draw(win)

    zone_text = Text(Point(.5, .08), "")
    zone_text.draw(win)

    total_score = 0
    stage_text = Text(Point(.5, .92), ("Round: "))
    stage_text.draw(win)

    wind_text = Text(Point(.5, .97), ("Wind: "))
    wind_text.draw(win)

    wsd = .25   # wind start deviation
    wd = .08    # wind deviation

    for stage in range(3):
        h_wind, v_wind = random.uniform(-wsd, wsd), random.uniform(-wsd, wsd)

        stage_text.setText("Round: {0}".format(stage + 1))

        for arrow in range(5):
            wth = .09    # wind threshold
            wind = ""

            if v_wind > wth:
                wind += "north"
            elif v_wind < -wth:
                wind += "south"

            if h_wind > wth:
                wind += "east"
            elif h_wind < -wth:
                wind += "west"

            if wind == "":
                wind = "Calm"

            wind_text.setText("Wind: {0}".format(wind.title()))

            cursor_pos = win.getMouse()
            arrow_x = cursor_pos.getX() + h_wind
            arrow_y = cursor_pos.getY() + v_wind

            h_wind += random.uniform(-wd, wd)
            v_wind += random.uniform(-wd, wd)

            arrow_pos = Circle(Point(arrow_x, arrow_y), .01)
            arrow_pos.setFill("black")

            arrow_zone = pract05.distance_between_points(Point(arrow_x, arrow_y), Point(.5, .5))

            if arrow_zone < .1:
                zone_text.setText("YELLOW ZONE (10 POINTS!)")
                score += 10
                arrow_pos.draw(win)
            elif arrow_zone >= .1 and arrow_zone < .2:
                zone_text.setText("RED ZONE (5 POINTS!)")
                score += 5
                arrow_pos.draw(win)
            elif arrow_zone >= .2 and arrow_zone <= .3:
                zone_text.setText("BLUE ZONE (2 POINTS!)")
                score += 2
                arrow_pos.draw(win)
            else:
                zone_text.setText("TARGET MISSED (0 POINTS!)")

            score_text.setText("Score: {0}".format(score))

        total_score += score
        score = 0

    avg_score = total_score / 3
    rounded_avg_score = round(avg_score)

    if rounded_avg_score > 44:
        grade = 0
    elif rounded_avg_score > 34:
        grade = 1
    elif rounded_avg_score > 24:
        grade = 2
    elif rounded_avg_score > 19:
        grade = 3
    else:
        grade = 4

    grade_text = grades[grade]
    stage_text.setText(grade_text)

    wind_text.setText("OVERALL AVERAGE SCORE: {0}".format(rounded_avg_score))
